PSEUDOCODE: Pet Adoption App

CLASS Pet:
    ATTRIBUTES:
        - name (String)
        - age (Integer)
        - breed (String)
    
    CONSTRUCTOR Pet(name, age, breed):
        SET this.name = name
        SET this.age = age
        SET this.breed = breed
    
    METHOD displayInfo():
        PRINT "Name: " + name + ", Age: " + age + ", Breed: " + breed

CLASS Dog EXTENDS Pet:
    ATTRIBUTES:
        - size (String)
    
    CONSTRUCTOR Dog(name, age, breed, size):
        CALL super(name, age, breed)
        SET this.size = size
    
    METHOD displayInfo():
        CALL super.displayInfo()
        PRINT "Size: " + size + ", Sound: Woof!"

CLASS Cat EXTENDS Pet:
    ATTRIBUTES:
        - isIndoor (Boolean)
    
    CONSTRUCTOR Cat(name, age, breed, isIndoor):
        CALL super(name, age, breed)
        SET this.isIndoor = isIndoor
    
    METHOD displayInfo():
        CALL super.displayInfo()
        PRINT "Indoor: " + isIndoor + ", Sound: Meow!"

CLASS Bird EXTENDS Pet:
    ATTRIBUTES:
        - canFly (Boolean)
    
    CONSTRUCTOR Bird(name, age, breed, canFly):
        CALL super(name, age, breed)
        SET this.canFly = canFly
    
    METHOD displayInfo():
        CALL super.displayInfo()
        PRINT "Can Fly: " + canFly + ", Sound: Tweet!"

MAIN METHOD:
    CREATE array pets with:
        - Dog("Buddy", 3, "Golden Retriever", "Large")
        - Cat("Whiskers", 2, "Persian", true)
        - Bird("Tweety", 1, "Canary", true)
    
    PRINT "=== Pet Adoption Center ==="
    FOR each pet in pets:
        CALL pet.displayInfo()
        PRINT newline
