PSEUDOCODE: Call Center Queue Management

CLASS Customer:
    ATTRIBUTES:
        - name (String)
        - priority (String) // "EMERGENCY" or "NORMAL"
        - callTime (Time)
    
    CONSTRUCTOR Customer(name, priority, callTime):
        SET this.name = name
        SET this.priority = priority
        SET this.callTime = callTime

CLASS CallCenter:
    ATTRIBUTES:
        - emergencyQueue (Queue of Customer)
        - normalQueue (Queue of Customer)
    
    CONSTRUCTOR CallCenter():
        INITIALIZE emergencyQueue as empty queue
        INITIALIZE normalQueue as empty queue
    
    METHOD addCustomer(customer):
        IF customer.priority equals "EMERGENCY":
            ENQUEUE customer to emergencyQueue
            PRINT "Emergency customer added: " + customer.name
        ELSE:
            ENQUEUE customer to normalQueue
            PRINT "Normal customer added: " + customer.name
    
    METHOD serveNextCustomer():
        IF emergencyQueue is not empty:
            SET customer = DEQUEUE from emergencyQueue
            PRINT "Serving emergency customer: " + customer.name
            RETURN customer
        ELSE IF normalQueue is not empty:
            SET customer = DEQUEUE from normalQueue
            PRINT "Serving normal customer: " + customer.name
            RETURN customer
        ELSE:
            PRINT "No customers waiting"
            RETURN null
    
    METHOD displayQueueStatus():
        PRINT "Emergency queue size: " + emergencyQueue.size()
        PRINT "Normal queue size: " + normalQueue.size()

MAIN METHOD:
    CREATE callCenter = new CallCenter()
    
    PRINT "Call Center Queue Management"
    
    CALL callCenter.addCustomer(Customer("John", "NORMAL", "10:00"))
    CALL callCenter.addCustomer(Customer("Jane", "EMERGENCY", "10:01"))
    CALL callCenter.addCustomer(Customer("Bob", "NORMAL", "10:02"))
    CALL callCenter.addCustomer(Customer("Alice", "EMERGENCY", "10:03"))
    
    CALL callCenter.displayQueueStatus()
    
    PRINT "\nServing customers:"
    CALL callCenter.serveNextCustomer()
    CALL callCenter.serveNextCustomer()
    CALL callCenter.serveNextCustomer()
    CALL callCenter.serveNextCustomer()
    CALL callCenter.serveNextCustomer()
