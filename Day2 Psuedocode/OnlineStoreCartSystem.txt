PSEUDOCODE: Online Store Cart System

CLASS Item:
    ATTRIBUTES:
        - id (String)
        - name (String)
        - price (Double)
        - quantity (Integer)
    
    CONSTRUCTOR Item(id, name, price, quantity):
        SET this.id = id
        SET this.name = name
        SET this.price = price
        SET this.quantity = quantity
    
    METHOD getTotalPrice():
        RETURN price * quantity
    
    METHOD toString():
        RETURN name + " (x" + quantity + ") - $" + price + " each = $" + getTotalPrice()

CLASS ShoppingCart<T extends Item>:
    ATTRIBUTES:
        - items (List<T>)
    
    CONSTRUCTOR ShoppingCart():
        INITIALIZE items as empty ArrayList<T>
    
    METHOD addItem(item):
        // Check if item already exists
        FOR each existingItem in items:
            IF existingItem.id equals item.id:
                SET existingItem.quantity += item.quantity
                PRINT "Updated quantity for: " + item.name
                RETURN
        
        // Add new item
        ADD item to items
        PRINT "Added to cart: " + item.name
    
    METHOD removeItem(itemId):
        FOR i = 0 to items.size()-1:
            IF items[i].id equals itemId:
                SET removedItem = REMOVE items[i]
                PRINT "Removed from cart: " + removedItem.name
                RETURN true
        PRINT "Item not found in cart"
        RETURN false
    
    GENERIC METHOD <T> calculateTotal():
        SET total = 0.0
        FOR each item in items:
            SET total += item.getTotalPrice()
        RETURN total
    
    METHOD displayCart():
        IF items is empty:
            PRINT "Cart is empty"
            RETURN
        
        PRINT "Shopping Cart Contents:"
        FOR each item in items:
            PRINT "  " + item.toString()
        PRINT "Total: $" + calculateTotal()
    
    METHOD getItemCount():
        SET count = 0
        FOR each item in items:
            SET count += item.quantity
        RETURN count

MAIN METHOD:
    CREATE cart = new ShoppingCart<Item>()
    
    PRINT "=== Online Store Cart System ==="
    
    // Add items to cart
    CALL cart.addItem(Item("I001", "Laptop", 999.99, 1))
    CALL cart.addItem(Item("I002", "Mouse", 25.50, 2))
    CALL cart.addItem(Item("I003", "Keyboard", 75.00, 1))
    CALL cart.addItem(Item("I001", "Laptop", 999.99, 1)) // Add another laptop
    
    PRINT newline
    CALL cart.displayCart()
    
    PRINT "\nTotal items in cart: " + cart.getItemCount()
    
    // Remove an item
    PRINT "\nRemoving mouse from cart..."
    CALL cart.removeItem("I002")
    
    PRINT newline
    CALL cart.displayCart()
